if (NOT DEFINED STEPIT_PLUGIN_MANIFEST_ID OR STEPIT_PLUGIN_MANIFEST_ID STREQUAL "")
  message(FATAL_ERROR "STEPIT_PLUGIN_MANIFEST_ID is required.")
endif ()

stepit_get_plugin_manifest_property(${STEPIT_PLUGIN_MANIFEST_ID} DIR STEPIT_PLUGIN_MANIFEST_DIR)
if (NOT DEFINED STEPIT_PLUGIN_MANIFEST_DIR OR STEPIT_PLUGIN_MANIFEST_DIR STREQUAL "")
  message(FATAL_ERROR "STEPIT_PLUGIN_MANIFEST_${STEPIT_PLUGIN_MANIFEST_ID}_DIR is required.")
endif ()

unset(STEPIT_PLUGIN_NAME)
unset(STEPIT_PLUGIN_DEPENDS)
set(STEPIT_PLUGIN_BUILDABLE TRUE)
set(STEPIT_PLUGIN_REASON "")

include("${STEPIT_PLUGIN_MANIFEST_DIR}/stepit_plugin_manifest.cmake")

if (NOT DEFINED STEPIT_PLUGIN_NAME OR STEPIT_PLUGIN_NAME STREQUAL "")
  message(
      FATAL_ERROR
      "Plugin manifest '${STEPIT_PLUGIN_MANIFEST_DIR}/stepit_plugin_manifest.cmake' "
      "must call stepit_declare_plugin(NAME <plugin> ...)."
  )
endif ()
if (NOT DEFINED STEPIT_PLUGIN_DEPENDS)
  set(STEPIT_PLUGIN_DEPENDS "")
endif ()
if (NOT DEFINED STEPIT_PLUGIN_BUILDABLE)
  set(STEPIT_PLUGIN_BUILDABLE TRUE)
endif ()
if (NOT DEFINED STEPIT_PLUGIN_REASON)
  set(STEPIT_PLUGIN_REASON "")
endif ()

stepit_set_plugin_manifest_property(${STEPIT_PLUGIN_MANIFEST_ID} NAME "${STEPIT_PLUGIN_NAME}")
stepit_set_plugin_manifest_property(${STEPIT_PLUGIN_MANIFEST_ID} DEPENDS ${STEPIT_PLUGIN_DEPENDS})
stepit_set_plugin_manifest_property(${STEPIT_PLUGIN_MANIFEST_ID} BUILDABLE ${STEPIT_PLUGIN_BUILDABLE})
stepit_set_plugin_manifest_property(${STEPIT_PLUGIN_MANIFEST_ID} REASON "${STEPIT_PLUGIN_REASON}")
